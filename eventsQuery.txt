-- BigQuery Query Instructions:
-- When running this query for the very first time, comment out the line that starts with "CREATE" to
-- first create the table. After that, comment out that line and uncomment the next line that
-- starts with "INSERT", which will insert new rows into that table each time this query is run. 
--
-- Note: We subtract 1 day from from the calcDate written to the table. For example, a query that runs early into 
-- October 30 will mostly have data that's from October 29. Ergo we subtract 1 from the calcDate so the table will show October 29
-- for the data collected over the prior 24 hours.

CREATE OR REPLACE TABLE `<your-gcp-project-id>.<yourDataSetName>.eventTallies` AS 
--INSERT `<your-gcp-project-id>.<yourDataSetName>.eventTallies` (date, eventName, country, eventSpecificDetailsJson, eventSpecificDetailsString, eventCount, userCount) 
SELECT DATE_ADD(DATE(DATETIME(calcDate)), INTERVAL -1 DAY) as date, eventName, country,
       PARSE_JSON(eventSpecificDetails) as eventSpecificDetailsJson, eventSpecificDetails as eventSpecificDetailsString,
       eventCount, userCount
FROM 
(SELECT jsonPayload.eventData.eventName as eventName,
        TimezoneToCountry(jsonPayload.eventData.timezone) as country,
        jsonPayload.eventData.eventSpecificDetails as eventSpecificDetails,
        COUNT(jsonPayload.eventData.eventName) AS eventCount, 
        COUNT(DISTINCT(jsonPayload.eventData.uid)) AS userCount, 
FROM `<your-gcp-project-id>.<yourDataSetName>.script_googleapis_com_console_logs_*`
WHERE 
      DATETIME(calcDate) >= DATETIME(timestamp)
AND
      DATETIME_DIFF(DATETIME(calcDate),
                    DATETIME(timestamp),
                    DAY) <= 1
GROUP BY eventName, country, eventSpecificDetails
ORDER BY eventName, country DESC
)
